// Filename: sw.js
const CACHE_NAME = 'filament-estimator-v1';

// List of files to cache (Tailwind and the placeholder icon included)
const ASSETS_TO_CACHE = [
  './Filament_calculator.html', 
  './manifest.json', 
  'https://cdn.tailwindcss.com', 
  'https://placehold.co/192x192/162447/ffffff?text=FILAMENT' 
];

// Installation: Caches all assets
self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => {
      return cache.addAll(ASSETS_TO_CACHE);
    })
  );
});

// Fetch: Serves from cache first, network second
self.addEventListener('fetch', (event) => {
  if (event.request.method !== 'GET') { return; }
  event.respondWith(
    caches.match(event.request).then((response) => {
      return response || fetch(event.request);
    })
  );
});

// Activation: Cleans up old caches
self.addEventListener('activate', (event) => {
  const cacheWhitelist = [CACHE_NAME];
  event.waitUntil(
    caches.keys().then((cacheNames) => {
      return Promise.all(
        cacheNames.map((cacheName) => {
          if (cacheWhitelist.indexOf(cacheName) === -1) {
            return caches.delete(cacheName);
          }
        })
      );
    })
  );
});
